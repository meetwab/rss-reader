```python
def create_subscription_manager():
    subscriptions = {}
    
    def add_subscription(name, url):
        if not name or not url:
            print("❌ 名称和URL都不能为空")
            return False
        
        if not url.startswith(('http://', 'https://')):
            print("❌ URL必须以http://或https://开头")
            return False
        
        subscriptions[name] = url
        print(f"✅ 成功添加订阅源: {name}")
        return True
    
    def remove_subscription(name):
        if name in subscriptions:
            del subscriptions[name]
            print(f"🗑️ 已删除订阅源: {name}")
            return True
        else:
            print(f"❌ 未找到订阅源: {name}")
            return False
    
    def list_subscriptions():
        if not subscriptions:
            print("📭 暂无订阅源")
            return
        
        print("📚 订阅源列表:")
        for i, (name, url) in enumerate(subscriptions.items(), 1):
            print(f"  [{i}] {name}: {url}")
    
    return {
        'add': add_subscription,
        'remove': remove_subscription,
        'list': list_subscriptions,
        'data': subscriptions
    }
```

```python
# app.py
from flask import Flask, render_template, request, jsonify
from rss_core import RSSManager

app = Flask(__name__)
rss_manager = RSSManager()

@app.route('/')
def index():
    """主页"""
    return render_template('index.html')

@app.route('/api/feeds', methods=['GET', 'POST'])
def handle_feeds():
    """RSS源管理API"""
    if request.method == 'GET':
        # 获取所有RSS源
        feeds = rss_manager.get_all_feeds()
        return jsonify(feeds)
    
    elif request.method == 'POST':
        # 添加新RSS源
        data = request.json
        name = data.get('name')
        url = data.get('url')
        
        try:
            result = rss_manager.add_feed(name, url)
            return jsonify({'success': True, 'message': '添加成功'})
        except Exception as e:
            return jsonify({'success': False, 'error': str(e)})

@app.route('/api/articles/<feed_name>')
def get_articles(feed_name):
    """获取指定RSS源的文章"""
    try:
        articles = rss_manager.get_articles(feed_name)
        return jsonify(articles)
    except Exception as e:
        return jsonify({'error': str(e)})

if __name__ == '__main__':
    app.run(debug=True)
```
